## Vue2
```js
const app = new Vue({
  el: '#app',
  data: {
    name: 'Leedom'
  }
})
```

## Vue3
```js
const app = {
  data() {
    return {
      name: 'Leedom'
    }
  }
}

Vue.createApp(app).mount('#app')
```
